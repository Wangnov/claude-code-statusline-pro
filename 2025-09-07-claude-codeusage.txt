  // 通过环境变量自动识别
  const PROXY_PATTERNS = {
    'co.yes.vg/team': 'yescode',
    'openrouter.ai': 'openrouter',
    // 更多中转站...
  };

  function detectProxy(): string | null {
    const baseUrl = process.env.ANTHROPIC_BASE_URL;
    if (!baseUrl) return null;

    for (const [pattern, provider] of Object.entries(PROXY_PATTERNS)) {
      if (baseUrl.includes(pattern)) return provider;
    }
    return 'custom'; // 未知中转站使用通用适配器
  }

  // src/proxy/engine.ts - 通用解析引擎
  interface ProxyConfig {
    name: string;
    endpoints: EndpointConfig[];
    display: DisplayConfig;
  }

  interface EndpointConfig {
    url: string;
    method: string;
    headers: Record<string, HeaderValue>;
    params?: Record<string, string>;
    extract: Record<string, string>;  // JSONPath表达式
  }

  class ProxyEngine {
    async fetchStats(config: ProxyConfig): Promise<ProxyData> {
      const results = {};

      for (const endpoint of config.endpoints) {
        // 1. 构建请求头
        const headers = this.buildHeaders(endpoint.headers);

        // 2. 发送请求
        const response = await fetch(endpoint.url, {
          method: endpoint.method,
          headers,
          params: endpoint.params
        });

        // 3. 提取数据
        const data = await response.json();
        for (const [key, path] of Object.entries(endpoint.extract)) {
          results[key] = this.extractValue(data, path);
        }
      }

      return results;
    }

    private extractValue(data: any, path: string): any {
      // 支持 JSONPath 和简单操作符
      // "summary[*].cost | sum" = 提取所有cost并求和
      // "data.total" = 直接路径
      // "requests | count" = 计数
    }
  }
